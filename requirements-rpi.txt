# ============================================================
# CORE DEPENDENCIES - ARM COMPATIBLE VERSIONS ONLY
# ============================================================

# NumPy: 1.24.3 causes "Illegal Instruction" on ARM
# Use 1.21.6 which is proven stable on RPi3
numpy==1.21.6

# OpenCV: install from apt for best ARMv7 compatibility
# sudo apt-get install -y python3-opencv

# TensorFlow Lite Runtime (Lightweight, ARM compatible)
tflite-runtime==2.13.0

# ============================================================
# DATABASE - ATTENTION: LanceDB not recommended for RPi3
# ============================================================
# PyArrow 13.0.0 has NO wheels for ARM - causes "Illegal Instruction"
# We're using SQLITE3 instead for better ARM compatibility
# LanceDB removed - use alternative: SQLite3 (included in Python)

# Utilities
Pillow==9.5.0

# ============================================================
# INSTALLATION COMMANDS FOR RASPBERRY PI 3
# ============================================================

# 2. sudo apt-get install python3.9 python3.9-dev python3-opencv libatlas-base-dev libjasper-dev
# 3. python3.9 -m pip install --upgrade pip setuptools wheel
# 4. python3.9 -m pip install -r requirements-rpi.txt --no-cache-dir
#

# - Full TensorFlow (too heavy)
# - ONNX Runtime (not for ARM)
